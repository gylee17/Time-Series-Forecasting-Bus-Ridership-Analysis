# Read the dataset
data <- read.csv("average-monthly-ridership.csv", stringsAsFactors = FALSE)
data$Month <- as.Date(paste0(data$Month, "-01"), format="%Y-%m-%d")

## Use xts package to divide the data into train (ending and including Dec 2018) and test (Jan - Jun 2019).

# Convert to xts object
ridership_xts <- xts(data[,2], order.by = data$Month)

# Split into train (ending and including Dec 2018) and test (Jan - Jun 2019)
train <- ridership_xts["/2018-12"]
test <- ridership_xts["2019-01/2019-06"]
## EDA of train data - qualitative analysis and indicate for each of the following if you think TS is stationary or not and why? 1. data plot 2.acf plot

# EDA - Plot train data
plot(train, main="Train Data: Average Monthly Bus Ridership", ylab="Ridership (in 100s)")

print("The time series is non-stationary because the plot shows an upward trend and the mean is changing over time.")
## [1] "The time series is non-stationary because the plot shows an upward trend and the mean is changing over time."
# ACF Plot
acf(coredata(train), main="ACF of Train Data")

print("The time series is non-stationary because the ACF values decrease gradually over many lags.")
## [1] "The time series is non-stationary because the ACF values decrease gradually over many lags."
## EDA of train data - quantitative analysis - does it support the qualitative analysis?

# Conduct ADF test for stationarity
adf_test <- adf.test(coredata(train))
print(adf_test)
## 
##  Augmented Dickey-Fuller Test
## 
## data:  coredata(train)
## Dickey-Fuller = -2.1874, Lag order = 4, p-value = 0.4989
## alternative hypothesis: stationary
print("Since the p-value is 0.4989 (> 0.05), we fail to reject the null hypothesis, meaning the data is non-stationary. This supports the qualitative analysis that the time series is non-stationary.")
## [1] "Since the p-value is 0.4989 (> 0.05), we fail to reject the null hypothesis, meaning the data is non-stationary. This supports the qualitative analysis that the time series is non-stationary."
## Use forecast::auto.arima with no seasonality to get a baseline model - what is the order (p, d, q) of the model?

# Baseline Model using auto.arima (no seasonality)
baseline_model <- auto.arima(train, seasonal=FALSE)
print(baseline_model)
## Series: train 
## ARIMA(0,1,2) with drift 
## 
## Coefficients:
##           ma1      ma2   drift
##       -0.0447  -0.3104  6.6540
## s.e.   0.0966   0.1053  3.3997
## 
## sigma^2 = 2997:  log likelihood = -578.7
## AIC=1165.4   AICc=1165.8   BIC=1176.09
print("The order of (p,d,q) is (0,1,2)")
## [1] "The order of (p,d,q) is (0,1,2)"
## Does the d support your EDA analysis for stationarity - why?

# Check if differencing supports stationarity analysis
d_value <- baseline_model$arma[6] # d value
cat("Differencing order (d):", d_value, "\n")
## Differencing order (d): 1
print("The ARIMA model selected by auto.arima has d = 1, meaning the algorithm determined that first-order differencing was necessary to make the series stationary.Therefore, the d supports the EDA analysis, confirming that the original data was non-stationary and needed transformation.")
## [1] "The ARIMA model selected by auto.arima has d = 1, meaning the algorithm determined that first-order differencing was necessary to make the series stationary.Therefore, the d supports the EDA analysis, confirming that the original data was non-stationary and needed transformation."
## Plot forecasts of the baseline model for 2019.

# Forecast for 2019 using baseline model
baseline_forecast <- forecast(baseline_model, h=6)
plot(baseline_forecast, main="Baseline Model Forecast for 2019")

## Run forecast: accuracy for the baseline model.

# Accuracy of baseline model
baseline_accuracy <- accuracy(baseline_forecast, test)
print(baseline_accuracy)

# Check ACF1 of residuals for each model
